   {activeTab === "PTO" && (
     <PTOTab employeeId={empId} userRole={role} />
   )}

3. ADD TO ADMINPANEL.JSX
   import LeaveAdminPanel from './components/LeaveAdminPanel';
   
   {activeAdminTab === "Leave" && <LeaveAdminPanel />}

4. USE IN CODE
   import { requestLeave, approveLeave } from './lib/useLeave';
   import { processAllEmployeeAccruals } from './lib/leaveAccrual';
   
   // Employee requests
   const req = await requestLeave(empId, typeId, start, end, reason);
   
   // Manager approves
   await approveLeave(req.id, managerId);
   
   // Admin processes annual accrual
   await processAllEmployeeAccruals(2025);
